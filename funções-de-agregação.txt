SELECT COUNT(*)
	FROM MEDICAMENTOS m 
WHERE M.ESTOQUE > 0

SELECT AVG(EXTRACT(YEAR FROM CURRENT_DATE) - EXTRACT(YEAR FROM DATA_NASCIMENTO)) AS MEDIA_IDADE
	FROM TUTOR t

SELECT MIN(m.PRECO) AS MENOR_VALOR
	FROM MEDICAMENTOS m
	
SELECT MAX(p.PESO) AS MAIOR_PESO
	FROM PACIENTE p
	
SELECT COUNT(*) AS TOTAL_TUTORES
	FROM PACIENTE p
	
SELECT m.ID_MEDICAMENTO, SUM (PRECO * ESTOQUE)
	FROM MEDICAMENTOS m
GROUP BY m.ID_MEDICAMENTO

SELECT i.ID_RECEITA,
SUM(i.TOTAL_ITEM)
	FROM ITENSRECEITA i 
GROUP BY i.ID_RECEITA
